# частотности для каждой буквы слова "most"
most_letters <- c("m", "o", "s", "t")

# преобразовываем таблицу в удобный формат с 3 столбцами: буква, язык, частотность
df_long <- df1 %>%
   pivot_longer(cols = -letter, names_to = "language", values_to = "frequency")

# оставляем только те ьуквы, которые нам нужны
letter_freqs <- df_long %>%
   filter(letter %in% most_letters) %>%
   pivot_wider(names_from = letter, values_from = frequency, values_fill = 0)

# априорная вероятность -- равная для каждого языка 1/13
prior = 1/nrow(letter_freqs)

# вычисляем для каждого языка
results <- letter_freqs %>%
   mutate(
      prior = prior,
      likelihood = m * o * s * t, # функция правдоподобия
      # просто перемножаем все вероятность относительно языка
      product = prior * likelihood, # числитель 
      posterior = product / sum(product) # апостериорная вероятность
   ) %>%
   select(language, prior, likelihood, product, posterior) %>%
   arrange(desc(posterior))

# находим наиболее вероятный язык
best_result <- results[which.max(results$posterior), c("language","posterior")]

print("Язык с макс. апостериорной вероятностью:")

минусы модели:

  • доля гласного находится в интервале от 0 до 1. А нормальное распределение обычно используют на данных, которые распределены от -∞ до +∞.

  • наши данные не совсем симметричны + мы видим аномально пустую области в районе 0.8-0.9

  • языки в датасете не независимы – то есть они могут разбиваться на отдельные семьи со схожими паттернами. Нормальная модель предполагает независимость наблюдений, что здесь может быть нарушено

плюсы модели:

  • одна из самых легких моделей с точки зрения математики (как для вычисление, так и для интерпретации)

  • если же мы все-таки предполагаем независимость всех событий, которые у нас представлены в датасете, тогда по центральной предельной теореме эти наблюдения будут стремиться к нормальному распределению …
